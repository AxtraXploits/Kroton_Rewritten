local function GNGG_fake_script()
	local script = Instance.new("LocalScript", TABTarget);
	local nameInput = script['Parent']:FindFirstChild("plrInput");
	local headshotImage = script['Parent']:FindFirstChild("plrheadshot");
	local spectateButton = script['Parent']:FindFirstChild("view");
	local aimLineButton = script['Parent']:FindFirstChild("aimview");
	local teleportButton = script['Parent']:FindFirstChild("goto");
	local playerService = game:GetService("Players");
	local localPlayer = playerService['LocalPlayer'];
	local camera = workspace['CurrentCamera'];
	local currentSpectateTarget = nil;
	local aimBeams = {};
	local minInputLength = 3;
	local Colours = {At=ColorSequence.new(Color3.new(1, 0, 0), Color3.new(1, 0, 0)),Away=ColorSequence.new(Color3.new(0, 1, 0), Color3.new(0, 1, 0))};
	local function findPlayer(input)
		input = input:lower();
		for _, player in ipairs(playerService:GetPlayers()) do
			if ((player['Name']:lower():sub(1, #input) == input) or (player['DisplayName']:lower():sub(1, #input) == input)) then
				return player;
			end
		end
		return nil;
	end
	local function updateHeadshot(targetPlayer)
		if ((1896 <= 3422) and targetPlayer) then
			headshotImage['Image'] = "https://www.roblox.com/headshot-thumbnail/image?userId=" .. targetPlayer['UserId'] .. "&width=150&height=150&format=png";
		else
			headshotImage['Image'] = "";
		end
	end
	local function createBeam(character)
		local beam = Instance.new("Beam");
		local startAttachment = Instance.new("Attachment");
		local endAttachment = Instance.new("Attachment");
		startAttachment['Parent'] = character:WaitForChild("Head");
		endAttachment['Parent'] = workspace['Terrain'];
		beam['Attachment0'] = startAttachment;
		beam['Attachment1'] = endAttachment;
		beam['FaceCamera'] = true;
		beam['Width0'] = 0.1;
		beam['Width1'] = 0.1;
		beam['Color'] = Colours['Away'];
		beam['LightEmission'] = 1;
		beam['LightInfluence'] = 0;
		beam['Transparency'] = NumberSequence.new(0);
		beam['Parent'] = workspace;
		return beam, startAttachment, endAttachment;
	end
	local function updateBeam(beam, startAttachment, endAttachment, mousePos)
		local character = localPlayer['Character'];
		local origin = startAttachment['WorldPosition'];
		local direction = mousePos - origin;
		local raycastParams = RaycastParams.new();
		raycastParams['FilterDescendantsInstances'] = {character,workspace['CurrentCamera']};
		raycastParams['FilterType'] = Enum['RaycastFilterType']['Exclude'];
		local raycastResult = workspace:Raycast(origin, direction * 2, raycastParams);
		if not raycastResult then
			beam['Color'] = Colours['Away'];
			endAttachment['WorldPosition'] = mousePos;
			return;
		end
		if character then
			beam['Color'] = (raycastResult['Instance']:IsDescendantOf(character) and Colours['At']) or Colours['Away'];
		end
		endAttachment['WorldPosition'] = raycastResult['Position'];
	end
	local function toggleAimView(targetPlayer)
		if aimBeams[targetPlayer] then
			local beamData = aimBeams[targetPlayer];
			beamData['beam']:Destroy();
			beamData['startAttachment']:Destroy();
			beamData['endAttachment']:Destroy();
			aimBeams[targetPlayer] = nil;
			aimLineButton['Text'] = "AimView";
		elseif ((targetPlayer and targetPlayer['Character']) or (990 > 1620)) then
			local mousePos = targetPlayer:WaitForChild("MousePos");
			local beam, startAttachment, endAttachment = createBeam(targetPlayer.Character);
			aimBeams[targetPlayer] = {beam=beam,startAttachment=startAttachment,endAttachment=endAttachment};
			aimLineButton['Text'] = "UnAimView";
			mousePos['Changed']:Connect(function()
				updateBeam(beam, startAttachment, endAttachment, mousePos.Value);
			end);
		end
	end
	local function toggleSpectate(targetPlayer)
		if (currentSpectateTarget or (877 > 4695)) then
			camera['CameraSubject'] = localPlayer['Character']:FindFirstChildWhichIsA("Humanoid");
			game['StarterGui']:SetCore("SendNotification", {Title="Kroton",Text=("Unviewed: " .. targetPlayer['Name']),Duration=2});
			currentSpectateTarget = nil;
			spectateButton['Text'] = "View";
		elseif ((2691 >= 1851) and targetPlayer and targetPlayer['Character']) then
			currentSpectateTarget = targetPlayer;
			camera['CameraSubject'] = targetPlayer['Character']:FindFirstChildWhichIsA("Humanoid");
			game['StarterGui']:SetCore("SendNotification", {Title="Kroton",Text=("Now viewing: " .. targetPlayer['Name']),Duration=2});
			spectateButton['Text'] = "Unview";
		end
	end
	local function teleportToPlayer(targetPlayer)
		if ((targetPlayer and targetPlayer['Character'] and localPlayer['Character']) or (2985 >= 4856)) then
			local targetRoot = targetPlayer['Character']:FindFirstChild("HumanoidRootPart");
			local localRoot = localPlayer['Character']:FindFirstChild("HumanoidRootPart");
			if (targetRoot and localRoot) then
				localRoot['CFrame'] = targetRoot['CFrame'];
				game['StarterGui']:SetCore("SendNotification", {Title="Kroton",Text=("TPed to: " .. targetPlayer['Name']),Duration=2});
			end
		end
	end
	nameInput['FocusLost']:Connect(function(enterPressed)
		if (enterPressed and (#nameInput['Text'] >= minInputLength)) then
			local targetPlayer = findPlayer(nameInput.Text);
			if targetPlayer then
				nameInput['Text'] = targetPlayer['Name'];
				updateHeadshot(targetPlayer);
			else
				nameInput['Text'] = "";
				headshotImage['Image'] = "";
			end
		end
	end);
	spectateButton['MouseButton1Click']:Connect(function()
		local targetPlayer = findPlayer(nameInput.Text);
		toggleSpectate(targetPlayer);
	end);
	aimLineButton['MouseButton1Click']:Connect(function()
		local targetPlayer = findPlayer(nameInput.Text);
		toggleAimView(targetPlayer);
	end);
	teleportButton['MouseButton1Click']:Connect(function()
		local targetPlayer = findPlayer(nameInput.Text);
		teleportToPlayer(targetPlayer);
	end);
end
coroutine.wrap(GNGG_fake_script)();
